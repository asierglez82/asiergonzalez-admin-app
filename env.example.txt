# Configuración para Social Media Credentials Manager y Gemini

# ========================================
# ALMACENAMIENTO DE CREDENCIALES
# ========================================

# Usar almacenamiento en la nube (Google Cloud Functions + Secret Manager)
# false = AsyncStorage local, true = Google Cloud
EXPO_PUBLIC_USE_CLOUD_STORAGE=false

# Habilitar publicación real en redes sociales desde Cloud Functions
# false = modo demo (no publica realmente), true = publica realmente
EXPO_PUBLIC_ENABLE_CLOUD_PUBLISH=false

# URL de la Cloud Function (solo si USE_CLOUD_STORAGE=true)
# Obtener después de desplegar: gcloud functions describe social-credentials --region=europe-west1 --format="value(serviceConfig.uri)"
EXPO_PUBLIC_CLOUD_FUNCTION_URL=https://europe-west1-tu-project-id.cloudfunctions.net/social-credentials

# ========================================
# GEMINI AI CONFIGURATION
# ========================================

# Usar servicio cloud de Gemini (Google Cloud Functions + Secret Manager)
# false = API key local, true = Google Cloud
EXPO_PUBLIC_USE_GEMINI_CLOUD=false

# URL de la Cloud Function de Gemini (solo si USE_GEMINI_CLOUD=true)
# Obtener después de desplegar: gcloud functions describe gemini-proxy --region=europe-west1 --format="value(serviceConfig.uri)"
EXPO_PUBLIC_GEMINI_CLOUD_FUNCTION_URL=https://europe-west1-tu-project-id.cloudfunctions.net/gemini-proxy

# API key local de Gemini (fallback y desarrollo)
# Obtener en: https://makersuite.google.com/app/apikey
EXPO_PUBLIC_GEMINI_API_KEY=tu-api-key-de-gemini

# Modelo por defecto
EXPO_PUBLIC_GEMINI_MODEL=gemini-2.5-flash

# ========================================
# GOOGLE CLOUD CONFIGURATION
# ========================================

# ID del proyecto de Google Cloud (opcional, se detecta automáticamente)
GOOGLE_CLOUD_PROJECT=tu-project-id

# Región de despliegue (recomendado: europe-west1 para Europa)
GOOGLE_CLOUD_REGION=europe-west1

# ========================================
# CONFIGURACIÓN DE DESARROLLO
# ========================================

# Habilitar logs de debug
NODE_ENV=development

# Timeout para requests (milisegundos)
EXPO_PUBLIC_REQUEST_TIMEOUT=30000

# ========================================
# CORS ORIGINS (para Cloud Functions)
# ========================================

# Dominios permitidos para CORS (separados por coma)
ALLOWED_ORIGINS=http://localhost:3000,https://tu-dominio.com
